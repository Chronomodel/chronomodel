111  {
112  mAdvanced->setWidget(mAdvancedWidget, 11*mMargin + 8*mLineH + mComboH + mDeltaHelp->heightForWidth(mWidth - 4*mMargin));

119 
120  // Disable methods forbidden by plugin
121 
122  const QStandardItemModel* model = qobject_cast<const QStandardItemModel*>(mMethodCombo->model());
123  for(int i=0; i<mMethodCombo->count(); ++i)
124  {
125  QStandardItem* item = model->item(i);
126  bool allowed = plugin->allowedDataMethods().contains((Date::DataMethod)i);
127 
128  item->setFlags(!allowed ? item->flags() & ~(Qt::ItemIsSelectable|Qt::ItemIsEnabled)
129  : Qt::ItemIsSelectable|Qt::ItemIsEnabled);
130  // visually disable by greying out - works only if combobox has been painted already and palette returns the wanted color
131  item->setData(!allowed ? mMethodCombo->palette().color(QPalette::Disabled, QPalette::Text)
132  : QVariant(), // clear item data in order to use default color
133  Qt::TextColorRole);
134  }

136  adaptSize();
