365 void ModelView::adjustStep()
366 {
367  Project* project = MainWindow::getInstance()->getProject();
368  QJsonObject state = project->state();
369  ProjectSettings s = ProjectSettings::fromJson(state[STATE_SETTINGS].toObject());
370

405 void ModelView::searchEvent()
406 {
407  QString search = mEventsSearchEdit->text();
408  QJsonObject state = MainWindow::getInstance()->getProject()->state();
409  QJsonArray events = state[STATE_EVENTS].toArray();
410 
411  int foundId;
412  QString bestName;
413  int bestScore = 99999999;
414 
415  int counter = 0;
416  for(int i=0; i<events.size(); ++i)
417  {
418  QJsonObject event = events[i].toObject();
419  int id = event[STATE_ID].toInt();
420  QString name = event[STATE_NAME].toString();
421 
422  int score = compareStrings(std::string(name.toUtf8()), std::string(search.toUtf8()));
423  if(abs(score) < bestScore)
424  {
425  bestScore = abs(score);
426  foundId = id;
427  bestName = name;
428  qDebug() << name;
429  }
430  ++counter;
431  }
432  qDebug() << "Names tried : " << counter << ", bestScore : " << bestScore;
433  mEventsScene->centerOnEvent(foundId);
434 }
