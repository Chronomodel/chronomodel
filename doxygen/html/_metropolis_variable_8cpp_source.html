43 float* MetropolisVariable::generateBufferForHisto(const QVector<double>& dataSrc, int numPts, double hFactor)
44 {
45  // Work with double precision here !
46  // Otherwise, "denum" can be very large and lead to infinity contribs!
47 
48  double sigma = dataStd(dataSrc);
49  if(sigma == 0)
50  return 0;
51 
52  double h = hFactor * 1.06 * sigma * pow(dataSrc.size(), -1./5.);
53 
54  double a = vector_min_value(dataSrc) - 4. * h;
55  double b = vector_max_value(dataSrc) + 4. * h;
56 
57  double delta = (b - a) / (numPts - 1);
58  double denum = delta * delta * dataSrc.size();
59 
60  float* input = (float*) fftwf_malloc(numPts * sizeof(float));
61  for(int i=0; i<numPts; ++i)
62  input[i]= 0.f;
63 
64  for(int i=0; i<(int)dataSrc.size(); ++i)
65  {
66  double t = dataSrc[i];
67 
68  double idx = (t - a) / delta;
69  double idx_under = floor(idx);
70  double idx_upper = idx_under + 1.;
71 
72  double contrib_under = (idx_upper - idx) / denum;
73  double contrib_upper = (idx - idx_under) / denum;
74 
75  if(std::isinf(contrib_under) || std::isinf(contrib_upper))
76  {
77  qDebug() << "FFT input : infinity contrib!";
78  }
79  if(idx_under < 0 || idx_under >= numPts || idx_upper < 0 || idx_upper > numPts)
80  {
81  qDebug() << "FFT input : Wrong index";
82  }
83  if(idx_under < numPts)
84  input[(int)idx_under] += (float)contrib_under;
85  if(idx_upper < numPts) // This is to handle the case when matching the last point index !
86  input[(int)idx_upper] += (float)contrib_upper;
87  }
88  return input;
89 }

119 QMap<double, double> MetropolisVariable::generateHisto(const QVector<double>& dataSrc, int fftLen, double hFactor, double tmin, double tmax)
120 {
121  int inputSize = fftLen;
122  int outputSize = 2 * (inputSize / 2 + 1);
123 
124  double sigma = dataStd(dataSrc);
125  double h = hFactor * 1.06 * sigma * pow(dataSrc.size(), -1.f/5.f);
126  double a = vector_min_value(dataSrc) - 4.f * h;
127  double b = vector_max_value(dataSrc) + 4.f * h;
128  double delta = (b - a) / fftLen;
129 
130  float* input = generateBufferForHisto(dataSrc, fftLen, hFactor);
131  float* output = (float*) fftwf_malloc(outputSize * sizeof(float));
132 
133  QMap<double, double> result;
134  if(input != 0)
135  {
136  // ----- FFT -----
137 
138  fftwf_plan plan_forward = fftwf_plan_dft_r2c_1d(inputSize, input, (fftwf_complex*)output, FFTW_ESTIMATE);
139  fftwf_execute(plan_forward);
140 
141  for(int i=0; i<outputSize/2; ++i)
142  {
143  double s = 2.f * M_PI * i / (b-a);
144  double factor = exp(-0.5f * s * s * h * h);
145 
146  output[2*i] *= factor;
147  output[2*i + 1] *= factor;
148  }
149 
150  fftwf_plan plan_backward = fftwf_plan_dft_c2r_1d(inputSize, (fftwf_complex*)output, input, FFTW_ESTIMATE);
151  fftwf_execute(plan_backward);
152 
153  // ----- FFT Buffer to result map -----
154 
155  for(int i=0; i<inputSize; ++i)
156  {
157  double t = a + (double)i * delta;
158  if(t >= tmin && t<= tmax)
159  {
160  result[t] = input[i];
161  }
162  }
163 
164  fftwf_free(input);
165  fftwf_free(output);
166  }
167 
168  return result;
169 }

225 void MetropolisVariable::generateCorrelations(const QList<Chain>& chains)
226 {
227  int hmax = 100;
228 
229  for(int c=0; c<chains.size(); ++c)
230  {
231  // Retourne la trace de la partie "acquisition" de la chaine :
232  QVector<double> trace = runTraceForChain(chains, c);
233 
234  // Correlation pour cette chaine
235  QVector<double> results;
236 
237  double n = trace.size();
238 
239  double sum = 0;
240  for(double i=0; i<n; ++i)
241  sum += trace[i];
242  double m = sum / n;
243 
244  double sum2 = 0;
245  for(double i=0; i<n; ++i)
246  sum2 += (trace[i] - m) * (trace[i] - m);
247 
248  for(double h=0; h<hmax; ++h)
249  {
250  double sumH = 0;
251  for(double i=0; i<n-h; ++i)
252  sumH += (trace[i] - m) * (trace[i + h] - m);
253 
254  double result = sumH / sum2;
255  results.append(result);
256  }
257  // Correlation ajoutée à la liste (une courbe de corrélation par chaine)
258  mCorrelations.append(results);
259  }
