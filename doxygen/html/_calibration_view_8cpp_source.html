232 void CalibrationView::updateZoom()
233 {
234  float min = mCalibGraph->minimumX();
235  float max = mCalibGraph->maximumX();
236  float minProp = 5 / (max - min);
237  float prop = (100. - mZoomSlider->value()) / 100.;
238  if(prop < minProp) prop = minProp;
239 
240  if(prop != 1)
241  {
242  // Remember old scroll position
243  double posProp = 0;
244  double rangeBefore = (double)mScrollBar->maximum();
245  if(rangeBefore > 0)
246  posProp = (double)mScrollBar->value() / rangeBefore;
247 
248  // Update Scroll Range
249  int fullScrollSteps = 1000;
250  int scrollSteps = (1.f - prop) * fullScrollSteps;
251  mScrollBar->setRange(0, scrollSteps);
252  mScrollBar->setPageStep(fullScrollSteps);
253 
254  // Set scroll to correct position
255  double pos = 0;
256  double rangeAfter = (double)mScrollBar->maximum();
257  if(rangeAfter > 0)
258  pos = floor(posProp * rangeAfter);
259  mScrollBar->setValue(pos);
260  }
261  else
262  {
263  mScrollBar->setRange(0, 0);
264  }
265  updateScroll();
266 }

282  double end = start + delta;
